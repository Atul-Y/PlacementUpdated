int longestSubarraySumAtMostK(vector<int>& nums, int k) {
    int i = 0, j = 0, sum = 0, maxLen = 0;

    while (j < nums.size()) {
        sum += nums[j];

        if (sum <= k) {
            maxLen = max(maxLen, j - i + 1);
            j++;
        } else {
            while (sum > k) {
                sum -= nums[i];
                i++;
            }
            j++;
        }
    }

    return maxLen;
}
